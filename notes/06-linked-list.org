bool hasCycle(ListNode* head) {
  ListNode* slow = head;
  ListNode* fast = head;

  while (fast && fast->next) {
    slow = slow->next;
    fast = fast->next->next;
    if (slow == fast) {
      return true;
    }
  }

  return false;
}

** 2.2 TRANSFORM
*** 2.2.1 REVERSE
#+NAME: 206. Reverse Linked List
#+BEGIN_SRC cpp
ListNode* reverse(ListNode* head) {
  ListNode* prev = nullptr;
  ListNode* curr = head;
  while (curr) {
    ListNode* nxt = curr->next;
    curr->next = prev;
    prev = curr;
    curr = nxt;
  }

  return prev;
}
#+END_SRC

** 2.3 COMBINE (two lists -> one)
*** 2.3.1 MERGE (by value, sorted)
#+NAME: 21. Merge Two Sorted Lists
#+BEGIN_SRC cpp
ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
  ListNode dummy(0);
  ListNode* curr = &dummy;

  while (list1 && list2) {
    if (list1->val < list2->val) {
      curr->next = list1;
      list1 = list1->next;
    } else {
      curr->next = list2;
      list2 = list2->next;
    }
    curr = curr->next;
  }

  curr->next = list1 ? list1 : list2;

  return dummy.next;
}
#+END_SRC

** 2.4 SPLIT (one list -> two)
*** 2.4.1 Find-Mid (fast-slow pointer)
#+NAME: 876. Middle of the Linked List
#+BEGIN_SRC cpp
ListNode* middleNode(ListNode* head) {
  ListNode* slow = head;
  ListNode* fast = head;
  while (fast && fast->next) {
    slow = slow->next;
    fast = fast->next->next;
  }
  return slow;
}
#+END_SRC
