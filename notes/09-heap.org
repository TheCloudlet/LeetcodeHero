** 1.2 [#A] Using C++ =std::priority_queue<>=
*The template declaration of std::priority_queue is non-trivial.*

- [[https://en.cppreference.com/w/cpp/container/priority_queue.html][cppreference - std::priority_queue]]

#+NAME: Snapshot from cppreference
#+BEGIN_SRC cpp
template<
    class T,
    class Container = std::vector<T>,
    class Compare = std::less<typename Container::value_type>
> class priority_queue;
#+END_SRC

- T :: typeof stored elements
- Container :: SequenceContainer, in leetcode use =std::vector<T>= most of the time
- *Compare* :: REMEMBER =std::less=, =return l < r=, smaller is at the lowest. Default is max-heap.

#+NAME: Example Usage
#+BEGIN_SRC
const auto data = {1, 4, 3, 2};

// (1) Pop: 1, 2, 3, 4
std::priority_queue<int, std::vector<int>, std::greater<int>>
    pq1(data.begin(), data.end());

// (2) Pop: 4, 3, 2, 1
struct {
  bool operator()(const int l, const int r) const { return l > r; };
} customLess;

std::priority_queue<int, std::vector<int>, customLess>
    pq2(data.begin(), data.end(), customLess);

// (3) Pop. 4, 3, 2, 1
auto cmp = [](int left, int right) {
  return left < right;
};
std::priority_queue<int, std::vector<int>, decltype(cmp)> pq3(cmp);

for (const int n : data) pq3.push(n);
#+END_SRC
